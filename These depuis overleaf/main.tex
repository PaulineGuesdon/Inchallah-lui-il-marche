\newcommand{\tocwithoutpagechapter}[1]{%
    \ifnum \c@tocdepth >\m@ne
        \addpenalty{-\@highpenalty}%
        \vskip 1.0em \@plus\p@
        \setlength\@tempdima{1.5em}%
        \begingroup
            \parindent \z@
            \rightskip \z@ \@plus 1fil
            \parfillskip \z@
            \leavevmode \bfseries
            \advance\leftskip\@tempdima
            \hskip -\leftskip
            #1\par
            \penalty\@highpenalty
        \endgroup
    \fi
}
\newcommand{\tocwithoutpagedotted}[4]{%
    \ifnum #1>\c@tocdepth \else
        \vskip \z@ \@plus.2\p@
        {\leftskip #2\relax \rightskip \z@ \@plus 1fil
         \parfillskip \z@
         \parindent #2\relax\@afterindenttrue
         \interlinepenalty\@M
         \leavevmode
         \@tempdima #3\relax
         \advance\leftskip \@tempdima \null\nobreak\hskip -\leftskip
         {#4}\par}%
    \fi
}
\renewcommand*\l@chapter[2]{%
    \begingroup
        \edef\currenttocpage{#2}%
        \ifx\currenttocpage\lasttocpage
            \tocwithoutpagechapter{#1}%
        \else
            \origl@chapter{#1}{#2}%
            \xdef\lasttocpage{#2}%
        \fi
    \endgroup
}
\renewcommand*\l@subsection[2]{%
    \begingroup
        \edef\currenttocpage{#2}%
        \ifx\currenttocpage\lasttocpage
            \tocwithoutpagedotted{2}{3.8em}{3.2em}{#1}%
        \else
            \origl@subsection{#1}{#2}%
            \xdef\lasttocpage{#2}%
        \fi
    \endgroup
}
\renewcommand*\l@subsubsection[2]{%
    \begingroup
        \edef\currenttocpage{#2}%
        \ifx\currenttocpage\lasttocpage
            \tocwithoutpagedotted{3}{7.0em}{4.1em}{#1}%
        \else
            \origl@subsubsection{#1}{#2}%
            \xdef\lasttocpage{#2}%
        \fi
    \endgroup
}
\renewcommand*\l@section[2]{%
    \begingroup
        \edef\currenttocpage{#2}%
        \ifx\currenttocpage\lasttocpage
            \tocwithoutpagedotted{1}{1.5em}{2.3em}{#1}%
        \else
            \origl@section{#1}{#2}%
            \xdef\lasttocpage{#2}%
        \fi
    \endgroup
}
